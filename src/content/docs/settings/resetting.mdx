import { Aside, } from '@astrojs/starlight/components';

You can view all the settings Starboard has to offer with the following command:
<CommandInput command="settings view" />

If you want a fresh start for your server, you can run the following command to fully reset your server's settings:
<CommandInput command="settings reset" options={[
    {
        type: OptionType.String,
        name: 'settings',
        value: 'Server Settings',
    }
]} />

If you just want to reset a singular setting back to its defaults, you can specify that setting to reset:
<CommandInput command="settings reset" options={[
    {
        type: OptionType.String,
        name: 'settings',
        value: 'Server Settings',
    },
    {
        type: OptionType.String,
        name: 'setting',
        value: 'Emojis',
    },
]} />
